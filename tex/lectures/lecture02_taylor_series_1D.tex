\begin{frame}{Local Polynomial Approximation}
In numerical analysis, we approximate functions locally using polynomials.

\vspace{0.75em}
The strategy is:
\begin{itemize}
  \item Assume a local polynomial model
  \item Determine coefficients by enforcing agreement at a point
  \item Use the result to construct numerical methods
\end{itemize}

\vspace{1em}
This approach underlies finite differences and finite elements.
\end{frame}

% -------------------------------------------------
\begin{frame}{Assumed Local Approximation}
We assume that near a point \(x_0\), the function \(f(x)\) can be approximated by
\[
f(x) \approx
d
+ a (x - x_0)
+ b (x - x_0)^2
+ c (x - x_0)^3.
\]

\vspace{1em}
The coefficients \(a\), \(b\), \(c\), and \(d\) are unknown
and will be determined by matching derivatives at \(x_0\).
\end{frame}

% -------------------------------------------------
\begin{frame}{Matching the Function Value}
Evaluate the approximation at \(x = x_0\):

\[
f(x_0) = d.
\]

\vspace{0.75em}
\[
\boxed{d = f(x_0)}
\]

\vspace{0.75em}
The constant term ensures the approximation is exact at the expansion point.
\end{frame}

% -------------------------------------------------
\begin{frame}{Matching the First Derivative}
Differentiate the approximation:
\[
\frac{d}{dx}
=
a + 2b(x - x_0) + 3c(x - x_0)^2.
\]

Evaluating at \(x = x_0\):
\[
f'(x_0) = a.
\]

\vspace{0.75em}
\[
\boxed{a = f'(x_0)}
\]
\end{frame}

% -------------------------------------------------
\begin{frame}{Matching the Second Derivative}
Differentiate again:
\[
\frac{d^2}{dx^2}
=
2b + 6c(x - x_0).
\]

Evaluating at \(x = x_0\):
\[
f''(x_0) = 2b.
\]

\vspace{0.75em}
\[
\boxed{b = \frac{1}{2} f''(x_0)}
\]
\end{frame}

% -------------------------------------------------
\begin{frame}{Matching the Third Derivative}
Differentiate a third time:
\[
\frac{d^3}{dx^3} = 6c.
\]

Evaluating at \(x = x_0\):
\[
f^{(3)}(x_0) = 6c.
\]

\vspace{0.75em}
\[
\boxed{c = \frac{1}{6} f^{(3)}(x_0)}
\]
\end{frame}

% -------------------------------------------------
\begin{frame}{Recovered Taylor Expansion}
Substituting the coefficients into the approximation:
\[
f(x) \approx
f(x_0)
+ f'(x_0)(x - x_0)
+ \frac{f''(x_0)}{2}(x - x_0)^2
+ \frac{f^{(3)}(x_0)}{6}(x - x_0)^3.
\]

\vspace{1em}
This is the Taylor expansion of \(f(x)\) about \(x_0\),
truncated after the cubic term.
\end{frame}

\begin{frame}{Taylor Expansion About the Center of the Interval}

\begin{equation*}
f(x) = \sin x + \frac{1}{10}x^2
\end{equation*}

\vspace{0.5em}

\vspace{0.5em}

\begin{block}{Key Observation}
When the expansion point \(x_0\) is centered in the interval \([a,b]\),
both the 2-term and 3-term Taylor approximations remain accurate over
a relatively wide region.
\end{block}

\end{frame}

\begin{frame}{Taylor Expansion About an Off-Center Point}

\begin{equation*}
f(x) = \sin x + \frac{1}{10}x^2
\end{equation*}

\vspace{0.5em}
\centering

\includegraphics[width=0.55\textwidth]{taylor_x0_minus1.png}

\vspace{0.5em}

\begin{block}{Key Observation}
When the expansion point \(x_0\) is not centered in the interval,
curvature effects become important and the 3-term Taylor approximation
is significantly more accurate near \(x_0\).
\end{block}

\end{frame}

% -------------------------------------------------
\begin{frame}{Key Takeaways}
\begin{itemize}
  \item Taylor series are constructed by matching derivatives at a point
  \item Each additional term enforces agreement of one higher derivative
  \item The constant term ensures exact agreement at the expansion point
  \item This construction will be reused to derive finite difference formulas
\end{itemize}
\end{frame}
